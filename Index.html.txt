<script>
  // ── STATE ──
  let data = JSON.parse(localStorage.getItem('gm_data') || 'null');
  if (!data) {
    data = { lists: [{ id: 'default', name: 'Main List', items: [] }], activeList: 'default' };
  }
  // migrate old items
  if (!data.lists) { data = { lists: [{ id: 'default', name: 'Main List', items: data.items || [] }], activeList: 'default' }; }

  let activeInsert = null;
  let lastAddedId = null;

  function save() { localStorage.setItem('gm_data', JSON.stringify(data)); }

  function getActive() { return data.lists.find(l => l.id === data.activeList) || data.lists[0]; }
  function getItems() { return getActive().items; }

  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg; t.classList.add('show');
    setTimeout(() => t.classList.remove('show'), 2500);
  }

  // ── LIST MANAGEMENT ──
  function openModal() {
    document.getElementById('new-list-name').value = '';
    document.getElementById('modal-overlay').classList.add('show');
    const delBtn = document.getElementById('delete-list-btn');
    delBtn.classList.toggle('show', data.lists.length > 1);
    setTimeout(() => document.getElementById('new-list-name').focus(), 100);
  }
  function closeModal() { document.getElementById('modal-overlay').classList.remove('show'); }
  function closeModalOutside(e) { if (e.target === document.getElementById('modal-overlay')) closeModal(); }

  function createList() {
    const name = document.getElementById('new-list-name').value.trim();
    if (!name) return;
    const id = 'list_' + Date.now();
    data.lists.push({ id, name, items: [] });
    data.activeList = id;
    save(); closeModal(); render();
    showToast('✅ ' + name + ' list உருவாக்கப்பட்டது!');
  }

  function switchList(id) {
    data.activeList = id;
    lastAddedId = null;
    activeInsert = null;
    save(); render();
  }

  function deleteCurrentList() {
    if (data.lists.length <= 1) { showToast('⚠️ கடைசி list delete செய்ய முடியாது!'); return; }
    if (!confirm(getActive().name + ' list delete ஆகும். சரியா?')) return;
    data.lists = data.lists.filter(l => l.id !== data.activeList);
    data.activeList = data.lists[0].id;
    save(); closeModal(); render();
    showToast('🗑️ List delete ஆச்சு');
  }

  // ── ITEMS ──
  function addItem() {
    const input = document.getElementById('new-item');
    const name = input.value.trim();
    if (!name) return;
    const items = getItems();
    if (items.find(i => i.name.toLowerCase() === name.toLowerCase())) {
      showToast('⚠️ இந்த item ஏற்கனவே இருக்கு!'); return;
    }
    const newItem = { id: Date.now(), name, selected: false, qty: 1 };
    items.push(newItem);
    lastAddedId = newItem.id;
    save(); input.value = ''; render();

    // update last added bar
    document.getElementById('last-added-bar').innerHTML = '⬆️ கடைசியா சேர்த்தது: <span>' + name + '</span>';
    showToast('✅ ' + name + ' சேர்க்கப்பட்டது!');

    // scroll to bottom of list
    setTimeout(() => {
      const el = document.getElementById('item-' + newItem.id);
      if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }, 100);
  }

  function insertAfter(afterId) {
    activeInsert = afterId; render();
    setTimeout(() => { const inp = document.getElementById('insert-input-' + afterId); if (inp) inp.focus(); }, 50);
  }

  function confirmInsert(afterId) {
    const inp = document.getElementById('insert-input-' + afterId);
    if (!inp) return;
    const name = inp.value.trim();
    if (!name) { cancelInsert(); return; }
    const items = getItems();
    if (items.find(i => i.name.toLowerCase() === name.toLowerCase())) { showToast('⚠️ இந்த item இருக்கு!'); return; }
    const idx = items.findIndex(i => i.id === afterId);
    const newItem = { id: Date.now(), name, selected: false, qty: 1 };
    items.splice(idx + 1, 0, newItem);
    lastAddedId = newItem.id;
    activeInsert = null; save(); render();
    showToast('✅ ' + name + ' இடையில் சேர்க்கப்பட்டது!');
  }

  function cancelInsert() { activeInsert = null; render(); }

  function toggleItem(id) {
    const item = getItems().find(i => i.id === id);
    if (item) { item.selected = !item.selected; save(); render(); }
  }

  function deleteItem(id) {
    getActive().items = getItems().filter(i => i.id !== id);
    if (lastAddedId === id) lastAddedId = null;
    save(); render(); showToast('🗑️ Item நீக்கப்பட்டது');
  }

  function changeQty(id, delta) {
    const item = getItems().find(i => i.id === id);
    if (item) { item.qty = Math.max(1, (item.qty || 1) + delta); save(); render(); }
  }

  function clearSelections() {
    if (!confirm('இந்த வாரம் selections clear ஆகும். சரியா?')) return;
    getItems().forEach(i => { i.selected = false; i.qty = 1; });
    save(); render(); showToast('✅ வார list clear ஆச்சு!');
  }

  // ── IMPORT / EXPORT ──
  function exportItems() {
    const items = getItems();
    if (!items.length) { showToast('⚠️ Items இல்ல!'); return; }
    const text = JSON.stringify(items);
    if (navigator.clipboard) {
      navigator.clipboard.writeText(text).then(() => showToast('📤 Export copy ஆச்சு!'));
    } else {
      const ta = document.createElement('textarea');
      ta.value = text; document.body.appendChild(ta); ta.select();
      document.execCommand('copy'); document.body.removeChild(ta);
      showToast('📤 Export copy ஆச்சு!');
    }
  }

  function toggleImport() {
    const area = document.getElementById('import-area');
    area.classList.toggle('show');
    if (area.classList.contains('show')) document.getElementById('import-text').focus();
  }

  function importItems() {
    const text = document.getElementById('import-text').value.trim();
    if (!text) { showToast('⚠️ Text paste பண்ணுங்கள்!'); return; }
    try {
      const imported = JSON.parse(text);
      if (!Array.isArray(imported)) throw new Error();
      const items = getItems(); let added = 0;
      imported.forEach(imp => {
        if (imp.name && !items.find(i => i.name.toLowerCase() === imp.name.toLowerCase())) {
          items.push({ id: Date.now() + Math.random(), name: imp.name, selected: false, qty: 1 });
          added++;
        }
      });
      save(); render();
      document.getElementById('import-text').value = '';
      document.getElementById('import-area').classList.remove('show');
      showToast('✅ ' + added + ' items import ஆச்சு!');
    } catch(e) { showToast('⚠️ தவறான data!'); }
  }

  // ── SHARE ──
  function shareList() {
    const selected = getItems().filter(i => i.selected);
    if (!selected.length) { showToast('⚠️ எந்த item-உம் select ஆகல!'); return; }
    const listName = getActive().name;
    const text = '🛒 ' + listName + ' - இந்த வாரம்:\n\n' + selected.map(i => i.name + '-' + i.qty).join('\n');
    if (navigator.share) { navigator.share({ title: listName, text }); }
    else { navigator.clipboard.writeText(text).then(() => showToast('📋 List copy ஆச்சு!')); }
  }

  // ── VIEW SWITCH ──
  function switchView(v) {
    document.querySelectorAll('.view-tab').forEach((t,i) => t.classList.toggle('active', (i===0&&v==='all')||(i===1&&v==='order')));
    document.getElementById('view-all').style.display = v==='all' ? 'block' : 'none';
    document.getElementById('view-order').style.display = v==='order' ? 'block' : 'none';
  }

  // ── ITEM HTML ──
  function itemHTML(item) {
    const isNew = item.id === lastAddedId;
    return `<div id="item-${item.id}">
      <div class="item-card ${item.selected ? 'selected' : ''} ${isNew ? 'just-added' : ''}" onclick="toggleItem(${item.id})">
        <div class="checkbox">${item.selected ? '✓' : ''}</div>
        <span class="item-name">${item.name}${isNew ? '<span class="new-tag">NEW</span>' : ''}</span>
        ${item.selected ? `<div class="item-qty" onclick="event.stopPropagation()">
          <div class="qty-btn" onclick="changeQty(${item.id},-1)">−</div>
          <span class="qty-num">${item.qty}</span>
          <div class="qty-btn" onclick="changeQty(${item.id},1)">+</div>
        </div>` : ''}
        <button class="delete-btn" onclick="event.stopPropagation();deleteItem(${item.id})">✕</button>
      </div>
      ${activeInsert === item.id ? `
      <div class="inline-insert show">
        <input id="insert-input-${item.id}" placeholder="இங்கே item type செய்யுங்க..." onkeydown="if(event.key==='Enter')confirmInsert(${item.id});if(event.key==='Escape')cancelInsert()">
        <button class="insert-confirm" onclick="confirmInsert(${item.id})">சேர்</button>
        <button class="insert-cancel" onclick="cancelInsert()">✕</button>
      </div>` : `
      <div class="insert-zone">
        <div class="insert-line"></div>
        <button class="insert-btn" onclick="insertAfter(${item.id})">+ இங்கே சேர்</button>
        <div class="insert-line"></div>
      </div>`}
    </div>`;
  }

  // ── RENDER ──
  function render() {
    const items = getItems();
    const selected = items.filter(i => i.selected);
    const unselected = items.filter(i => !i.selected);

    document.getElementById('all-count').textContent = items.length;
    document.getElementById('order-count').textContent = selected.length;

    // list tabs
    const tabsEl = document.getElementById('list-tabs');
    tabsEl.innerHTML = data.lists.map(l => `
      <button class="list-tab ${l.id === data.activeList ? 'active' : ''}" onclick="switchList('${l.id}')">${l.name}</button>
    `).join('') + `<button class="list-tab-add" onclick="openModal()">+ புதிய List</button>`;

    // items list
    const listEl = document.getElementById('items-list');
    if (!items.length) {
      listEl.innerHTML = `<div class="empty"><div class="icon">🛍️</div><p>Items இல்ல!<br>மேலே type செய்து சேர்க்கவும்</p></div>`;
    } else {
      let html = '';
      if (selected.length) {
        html += `<div class="list-label">✅ இந்த வாரம் (${selected.length})</div>`;
        html += selected.map(itemHTML).join('');
      }
      if (unselected.length) {
        html += `<div class="list-label">📦 எல்லா items (${unselected.length})</div>`;
        html += unselected.map(itemHTML).join('');
      }
      listEl.innerHTML = html;
    }

    // order list
    const orderEl = document.getElementById('order-list');
    if (!selected.length) {
      orderEl.innerHTML = `<div class="empty"><div class="icon">🗂️</div><p>எந்த item-உம் select ஆகல!<br>"எல்லாமே" tab-ல் போய் tick செய்யுங்க</p></div>`;
    } else {
      let html = `<div class="order-hero"><h2>${getActive().name}</h2><span>${selected.length} items ready ✓</span></div>`;
      html += selected.map(item => `<div class="order-item">
        <div class="order-item-left"><div class="order-dot"></div><span class="order-name">${item.name}</span></div>
        <span class="order-qty">- ${item.qty}</span></div>`).join('');
      html += `<button class="share-btn" onclick="shareList()">📤 WhatsApp-ல் Share செய்</button>`;
      html += `<button class="clear-btn" onclick="clearSelections()">🔄 வாரம் முடிஞ்சது — Clear செய்</button>`;
      orderEl.innerHTML = html;
    }
  }

  render();
